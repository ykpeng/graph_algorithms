def add_edge(u, v, options = {})
  if !@nodes[u]
    @adj[u] = {}
    @nodes[u] = {}
  end
  if !@nodes[v]
    @adj[v] = {}
    @nodes[v] = {}
  end
  data = @adj[u][v] || {}
  data.merge!(options)
  @adj[u][v] = data
  @adj[v][u] = data
end

def add_node(node, options = {})
  if @nodes[node]
    @nodes[node].merge!(options)
  else
    @nodes[node] = options
    @adj[node] = {}
  end
end

def remove_edge(u, v)
  if @adj[u][v]
    @adj[u].delete(v)
    if u != v
      @adj[v].delete(u)
    end
  else
    raise NetworkrError, "Edge #{u}-#{v} not in graph"
  end
end

def remove_node(node)
  if @nodes[node]
    @nodes.delete(node)
  else
    raise NetworkrError, "Node #{node} not in graph"
  end

  nbrs = @adj[node]
  nbrs.each_key do |nbr|
    @adj[nbr].delete(node)
  end
  @adj.delete(node)
end

def to_s
  res = ''
  @adj.each do |u, nbrs|
    nbrs.each do |v, options|
      res = "#{res}#{u}-#{v} #{options}\n"
    end
  end
  res[0...-1]
end
